3(a)

clear
fig5 = figure(5);
Vmax = 20; % mM/s
K = 50; % mM
h = 4;

deltaT = 0.01; % s
maxT = 10; % s
numiterations = maxT/deltaT;
t = 0:deltaT:maxT;
S0 = 100; % initial concentration (mM)
S = zeros(size(t));
S(1) = S0;
for i = 1:numiterations
    dSdt = -(Vmax*(S(i)^h))/((K^h) + (S(i)^h));
    S(i+1) = S(i) + dSdt*deltaT;
end
plot(t, S)
xlabel('Time (s)')
ylabel('Substrate Concentration (mM)')
